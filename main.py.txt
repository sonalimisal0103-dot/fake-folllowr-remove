from instagrapi import Client
import time
import json

print("🔥 Instagram Fake Followers Remover Tool")

username = input("theharsh_.01: ")
password = input("HARSHAD00: ")

cl = Client()
cl.login(username, password)
print("✅ Login Successful!")

user_id = cl.user_id_from_username(username)
print(f"Your User ID: {user_id}")

# Get all followers
print("Fetching followers... (ye thoda time lega)")
followers = cl.user_followers(user_id, amount=0)  # 0 = all

print(f"Total Followers: {len(followers)}")

fake_list = []
suspicious_list = []

for user_id, user in followers.items():
    username = user.username
    full_name = user.full_name
    is_private = user.is_private
    is_verified = user.is_verified
    follower_count = user.follower_count
    following_count = user.following_count
    media_count = user.media_count
    profile_pic = user.profile_pic_url

    score = 0
    reasons = []

    if not profile_pic or "default" in str(profile_pic):
        score += 3
        reasons.append("No Profile Pic")

    if follower_count < 10 and following_count > 50:
        score += 2
        reasons.append("Low Followers, High Following")

    if media_count == 0:
        score += 2
        reasons.append("No Posts")

    if follower_count == 0 and following_count > 100:
        score += 3
        reasons.append("Ghost Account")

    if score >= 4:
        fake_list.append({
            "username": username,
            "score": score,
            "reasons": reasons
        })
    elif score >= 2:
        suspicious_list.append({
            "username": username,
            "score": score,
            "reasons": reasons
        })

# Results
print("\n" + "="*50)
print(f"🚨 FAKE / SUSPICIOUS ACCOUNTS FOUND: {len(fake_list) + len(suspicious_list)}")
print("="*50)

print(f"\n🔴 High Risk Fake Accounts ({len(fake_list)}):")
for acc in fake_list[:30]:  # Top 30
    print(f"@{acc['username']} | Score: {acc['score']} | Reasons: {', '.join(acc['reasons'])}")

print(f"\n🟡 Suspicious Accounts ({len(suspicious_list)}):")
for acc in suspicious_list[:20]:
    print(f"@{acc['username']} | Score: {acc['score']} | Reasons: {', '.join(acc['reasons'])}")

# Save to file
with open("fake_followers.json", "w", encoding="utf-8") as f:
    json.dump({"fake": fake_list, "suspicious": suspicious_list}, f, indent=4, ensure_ascii=False)

print("\n✅ Results saved in 'fake_followers.json'")
